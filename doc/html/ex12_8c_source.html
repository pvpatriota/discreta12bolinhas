330 void criar_threads(tadt **p_threads, transicoes *p_transicoes, estados *p_estados)
331 {
332  transicoes *pl=p_transicoes;
333  tadt *pt, *plant=NULL;

335  {

339  pt->std=p_estados;
340  if(plant!=NULL)
341  plant->prox=pt;
342  else
343  *p_threads=pt;
344  plant=pt;
345  pt=pt->prox;
346  pl=pl->prox;
347  }
348  pt=*p_threads; /*colocando ponteiro auxiliar como cabeca da lista em questao.*/
349  if(DEBUG)
350  {
351  printf("*****Lista de threads******.\n");
352  while(pt!=NULL)
353  {
354  printf("thread da transicao %d.\n", pt->tr->ntr);
355  pt=pt->prox;
356  }
357  pt=*p_threads; /*Colocando o ponteiro auxiliar como cabeca novamente caso o debug seja utilizado.*/
358  printf("*****Lista de threads******.\n");
359  }
360  while(pt!=NULL)
361  {
362  if(DEBUG)
363  printf("Thread da transicao %d sendo criada.\n", pt->tr->ntr);
364  pthread_create(&(pt->nth), NULL, roda_thread, (void *)pt);

396 void *roda_thread(void *dados)
397 {
398  int cont=1, aux;
399  tadt *pl=(tadt *)dados;
400  arcos *pa;
401  estados *pe;
402  while(cont<=ITERACOES) /*A quantidade de iteracoes pode ser modificada pela alteracao do define.*/
403  {
404  aux=1; /*Variavel que vai ser utilizada como forma de checar a possibilidade, ou nao, da ativacao da transicao.*/
405  pa=pl->tr->entram;
406  while(pa!=NULL)
407  {
408  pe=procurar_estado(pl->std, pa->origem);
409  if(pa->custo>pe->nt)
410  aux=0; /*A ativacao vai ser considerada sempre verdadeira, sendo considerada falsa apenas caso esteja faltando algum token.*/
411  pa=pa->prox;
412  }
413  if(aux) /*Sera ativada apenas se todos os estados de origem da transicao estiverem com mais tokens que o necessario.*/
414  {
415  pa=pl->tr->entram;
416  while(pa!=NULL)
417  {
418  pe=procurar_estado(pl->std, pa->origem);
419  pe->nt-=pa->custo;
420  pa=pa->prox;
421  }
422  pa=pl->tr->saem;
423  while(pa!=NULL)
424  {
425  pe=procurar_estado(pl->std, pa->destino);
426  pe->nt+=pa->custo;
427  pa=pa->prox;
428  }
429  }
430  cont++;
431  }
432  if(DEBUG)
433  printf("Thread da transicao %d reportando.\n", pl->tr->ntr);
434  return NULL;
435 }

684 void desenha_arcos(int qo, int qf, BITMAP *buff, int k, int c, int flag)
685 {
686  float si,co, alfa,beta, phi, x1, y1, x2, y2, x3, y3, xo, yo, xf, yf, raio, xt1, yt1, xt2, yt2, rc;
687  raio=(Y/8)*(M_PI/(M_PI+(k/2)));
688  rc = YCentro - raio*4;
689  y1 = YCentro + rc*cos((2*M_PI/k)*qo);
690  x1 = XCentro + rc*sin((2*M_PI/k)*qo);
691  y3 = YCentro + rc*cos((2*M_PI/k)*qf);
692  x3 = XCentro + rc*sin((2*M_PI/k)*qf);
693  alfa=arctan(x1,y1,x3,y3);
694  y2=(y3+y1)/2 + raio * cos(alfa);
695  x2=(x3+x1)/2 - raio * sin(alfa);
696 
697  if(((alfa >= 0) && (alfa <= M_PI/2)) || ((alfa >= M_PI) && (alfa <= 3*M_PI/2)))
698  {
699  beta=arctan(x3,y3,x2,y2);
700  phi=arctan(x1,y1,x2,y2);
701  if(flag)
702  {
703  xo = x1 + raio * cos(phi);
704  yo = y1 + raio * sin(phi);
705  xf = x3;
706  yf = y3;
707  }
708  else
709  {
710  xo = x1;
711  yo = y1;
712  xf = x3 + raio * cos(beta);
713  yf = y3 + raio * sin(beta);
714  }
715  }
716  else
717  {
718  alfa=arctan(x3,y3,x1,y1);
719  y2=(y3+y1)/2 + raio * cos(alfa);
720  x2=(x3+x1)/2 - raio * sin(alfa);
721  beta=arctan(x1,y1,x2,y2);
722  phi=arctan(x3,y3,x2,y2);
723  if(flag)
724  {
725  xo = x1 - raio * cos(phi);
726  yo = y1 - raio * sin(phi);
727  xf = x3;
728  yf = y3;
729  }
730  else
731  {
732  xo = x1;
733  yo = y1;
734  xf = x3 - raio * cos(beta);
735  yf = y3 - raio * sin(beta);
736  }
737  }
738  int coo[8];
739  coo[0] = (int)xo;
740  coo[1] = (int)yo;
741  coo[2] = (int)x2;
742  coo[3] = (int)y2;
743  coo[4] = (int)x2;
744  coo[5] = (int)y2;
745  coo[6] = (int)xf;
746  coo[7] = (int)yf;
747  spline(buff,coo,CORBRANCO);
748 
749  si=lsin(x2,y2,xf,yf);
750  co=lcos(x2,y2,xf,yf);
751  xt1 = xf - (raio / 4) * (si + co);
752  yt1 = yf + (raio / 4) * (co - si);
753  xt2 = xf + (raio / 4) * (si - co);
754  yt2 = yf - (raio / 4) * (si + co);
755  triangle(buff, xf, yf, xt1, yt1, xt2, yt2, CORBRANCO);
756  textprintf_ex(buff, font, x2, y2, CORVERDE, CORPRETO, "%d", c);
757 
758 }

771 {
772  if(x1==0 && x2 == 0 && y1 == 0 && y2 == 0)
773  return ~0;
774  return ((x2-x1)/sqrt((x2-x1)*(x2-x1)+(y2-y1)*(y2-y1)));
